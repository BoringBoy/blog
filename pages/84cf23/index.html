# test 判断表达式

# 基本语法

#常规写法
if test commands
then
  commands
else
  commands
fi

#或使用'[]'符号代替
if [command]

  commands
else
  commands

# 文件比较

(1) 判断文件是否存在

if [ -e $file]

(2)判断文件存在并且是目录

if [ -d $file]

(3)判断file是文件

if [ -f $file ]

(4)判断文件存在且可读

if [ -r $file ]

(5)判断文件存在且可写

if [ -w $file ]

(6)判断文件存在且可执行

if [ -x $file ]

(7)判断文件存在并且非空

if [ -s $file ]

(8)判断文件存在且归当前用户所有

if [ -o $file ]

(9)判断文件存在且默认组与当前用户一致

if [ -g $file ]
